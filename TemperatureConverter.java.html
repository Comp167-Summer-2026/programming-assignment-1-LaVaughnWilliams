public static void main(String[] args) {
    Scanner input = new Scanner(System.in);
    String errorHandled;
    String decision = "";
    double temperatureC;
    double temperatureF;
    while (decision.isEmpty()) {
        System.out.print("Enter a temperature variant(c or f) or type stop: ");
        if (input.hasNext("c")) {
            System.out.print("enter a temperature");
            temperatureC = input.nextDouble();
        } else if (input.hasNext("f")) {
            System.out.print("enter a temperature");
            temperatureF = input.nextDouble();
            System.out.print(temperatureF);
        } else if (input.hasNext("stop")) {
            System.out.print("ending program");
            System.exit(0);
        }
        else {
            System.out.println("Invalid input, please try again");

    }
